JavaScript 代码

// ES6 异步函数示例
const fetchData = async (url) => {
    try {
        const response = await fetch(url);
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('获取数据失败:', error);
        throw error;
    }
};

// 使用示例
fetchData('/api/agents')
    .then(data => console.log(data))
    .catch(err => console.error(err));
